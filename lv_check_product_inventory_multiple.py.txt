class ProductGet:

        pass

    def readToCSV(self, csvFile):
        csvData = []

        with open(csvFile, 'r', encoding='utf-8', newline='') as CSV:
            read = csv.reader(CSV)
            # header = next(read)

            for row in read:
                csvData.append(row[0])

        return csvData

    def __init__(self):

        self.commonData = {

            'longitudeCenter': '126.9',

            'longitudeA': '124.6',
            'latitudeB': '33.2',
            'longitudeB': '129.6',
            'query': 'kr'
        }

    def allStoresQuery(self):
        allStoreList = []
        seoulStoreList = []

        queryData = {
            'pageType': 'store_locator'
        }

        max_retry = 3
        retry_delay = 5  # 초 단위
        attempt = 0
        queryResponse = None

        while attempt < max_retry:
            queryResponse = requests.post(
                url = self.commonURL,
                cookies = self.commonCookie,
                headers = self.commonHeader,
                json = self.commonData | queryData
            )
            if queryResponse.status_code == 200:
                break
            elif queryResponse.status_code in [500, 504]:
                print(f'⚠️ {queryResponse.status_code} Gateway Timeout in allStoresQuery - {retry_delay}초 후 재시도 ({attempt+1}/{max_retry})')
                time.sleep(retry_delay)
                attempt += 1
            else:
                print(f'❌ Unhandled status code in allStoresQuery: {queryResponse.status_code}')
                print(f'Response: {queryResponse.text}')
                return [], [] # Return empty lists on unhandled error

        if queryResponse is None or queryResponse.status_code != 200:
            print("❌ allStoresQuery failed after retries.")
            return [], []

        # print(f'{queryResponse.text}') #### TEST PRINT ####
        queryResponseDict = json.loads(queryResponse.text)
        # print(queryResponseDict) #### TEST PRINT ####

        if 'hits' not in queryResponseDict:
            print("Error: 'hits' key not found in API response for allStoresQuery.")
            print("Response text:", queryResponse.text)
            return [], []

        for hits in queryResponseDict['hits']:
            allStoreList.append([f'{hits["name"]}', hits["geo"]["latitude"], hits["geo"]["longitude"]])
        
        allStoreList = sorted(allStoreList, key=lambda x : (x[1], -x[2]), reverse=True)

        for c in allStoreList:
            if c[1] >= 37.45 and c[2] <= 127.18:
                seoulStoreList.append(c)

        return allStoreList, seoulStoreList

    def prdStoresQuery(self, num, sku):
        num += 1

        storeAvailList[0].append(sku)
        storeAvailList[0].append([])

        queryData = {
            'flagShip': False,
            'clickAndCollect': False,
            'skuId' : sku,
            'pageType': 'productsheet'
        }

        queryResponse = requests.post(
            url = self.commonURL,
            cookies = self.commonCookie,
            headers = self.commonHeader,
            json = self.commonData | queryData
        )

        max_retry = 3
        retry_delay = 5  # 초 단위
        attempt = 0

        while attempt < max_retry:
            queryResponse = requests.post(
                url = self.commonURL,
                cookies = self.commonCookie,
                headers = self.commonHeader,
                json = self.commonData | queryData
            )
            # print(f'[SKU {sku}] 응답 상태 코드: {queryResponse.status_code}')

            if queryResponse.status_code == 200:
                break
            elif queryResponse.status_code == 500:
                print(f'⚠️ 500 Gateway Timeout 발생 - {retry_delay}초 후 재시도 ({attempt+1}/{max_retry})')
                time.sleep(retry_delay)
                attempt += 1
            elif queryResponse.status_code == 504:
                print(f'⚠️ 504 Gateway Timeout 발생 - {retry_delay}초 후 재시도 ({attempt+1}/{max_retry})')
                time.sleep(retry_delay)
                attempt += 1
            else:
                print(f'❌ 처리 불가 상태 코드: {queryResponse.status_code}')
                return storeAvailList  # 실패한 경우라도 형식 유지해서 반환

        # 반복문 탈출 후 처리
        try:
            storeList = json.loads(queryResponse.text)
        except json.JSONDecodeError as e:
            print(f'❌ JSON 파싱 오류: {e}')
            return storeAvailList

        for h in storeList['hits']:
            #print(f'{n2+1}: ' + h.get('name'), end=': ')
            for t in h['additionalProperty']:
                if t['name'] == 'stockAvailability':
                    if t['value'] == 'true':
                        pass
                        #print(n1, h.get('name'), f'O', end=', ')
                        storeAvailList[0][2].append([h.get('name'), f'O'])
                    elif t['value'] == 'false':
                        pass
                        #print(n1, h.get('name'), f'X', end=', ')
                        storeAvailList[0][2].append([h.get('name'), f'X'])
                    else:
                        pass
                        #print(n1, h.get('name'), f'?', end=', ')
                        storeAvailList[0][2].append([h.get('name'), f'?'])

        #for n1 in range(len(storeAvailList[0])):

        #        print(f'{storeAvailList[0][2][n2][0]}', end=', ')
        #        print(f'{storeAvailList[0][2][n2][1]}')

        return storeAvailList

class ExportExcel:

        wb = xl.load_workbook(self.filepath)
        ws1 = wb[self.ws1.title]

        cols = 3
        
        #print(ws1.cell(1, 3).value)
        for i in range (ws1.max_column):

                ws1.cell(1, i + 1, re.findall(r'^루이\s비통\s(.*)', ws1.cell(1, i + 1).value)[0])

        wb.save(self.filepath)
        wb.close()

def main():

        exit()

    p = ProductGet()
    prdSKU = p.readToCSV(csvFile)

    try:
        filepath = 'LOUIS_VUITTON'
        timestamp = int(time.time())

        a = StoreQueryAPI()
        r = ExportExcel(filepath, timestamp)

        allStoreList, seoulStoreList = a.allStoresQuery()

        for num, sku in enumerate(prdSKU):
